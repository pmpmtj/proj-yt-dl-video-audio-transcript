# Run all tests
python -m pytest

# Run all tests specific test file
python -m pytest tests/test_yt_dl_core.py

# Run with verbose output (shows each test name)
python -m pytest tests/test_yt_dl_core.py -v

# Run specific test method
python -m pytest tests/test_yt_dl_core.py::TestDownloadAudioMP3::test_download_audio_mp3_file_already_exists -v
 
 ---

# Run only video tests
python -m pytest tests/test_yt_dl_core.py::TestDownloadVideoWithAudio -v

# Run specific test
python -m pytest tests/test_yt_dl_core.py::TestDownloadVideoWithAudio::test_download_video_with_audio_custom_parameters -v


## Useful Options:

# Stop on first failure
python -m pytest -x

# Show print statements
python -m pytest -s

# Run tests in parallel (if you have many tests)
python -m pytest -n auto

# Generate coverage report
python -m pytest --cov=src
